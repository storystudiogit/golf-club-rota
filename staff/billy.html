    <script>
    const DAYS = ['Mon','Tue','Wed','Thu','Fri','Sat','Sun'];
    const SHIFTS = ['06:00-12:00','12:00-17:00','17:00-22:00'];
    const SHIFT_HRS = {'06:00-12:00':6,'12:00-17:00':5,'17:00-22:00':5};
    function jobColour(job) {
    const map = {
    'BAR':'#FFC7CE','BAR #2':'#FFCCAA','GREEN FEES':'#C6EFCE',
    'GREEN FEES #2':'#B8E4B8','JUNGLE FALLS':'#FFEB9C',
    'RANGE':'#BDD7EE','SITE GENERAL':'#E2CFFF','STARTER':'#FCE4D6'
    };
    return map[job] || '#f0f0f0';
    }
    // Get staff name from URL: staff/caz.html → CAZ
    const pathParts = window.location.pathname.split('/');
    const filename = pathParts[pathParts.length - 1].replace('.html','');
    const staffName = filename.replace(/-/g,' ').toUpperCase();
    document.getElementById('staff-name').textContent = staffName;
    document.title = staffName + ' — Schedule';
    async function load() {
    try {
    const res = await fetch('../data.json?v=' + Date.now());
    if (!res.ok) throw new Error();
    const data = await res.json();
    render(data, staffName);
    } catch(e) {
    document.getElementById('loading').style.display = 'none';
    document.getElementById('error').style.display = 'block';
    }
    }
    function render(data, name) {
    document.getElementById('loading').style.display = 'none';
    document.getElementById('content').style.display = 'block';
    document.getElementById('week-label').textContent = data.weekLabel || '';
    document.getElementById('footer').textContent = 'Last published: ' + (data.lastUpdated || '');
    // Build per-day, per-shift view for this staff member
    let totalHours = 0;
    let workingDays = 0;
    const container = document.getElementById('days-container');
    DAYS.forEach(day => {
    let dayHours = 0;
    const shifts = [];
    SHIFTS.forEach(shift => {
    let assignedJob = null;
    Object.entries(data.rota || {}).forEach(([job, jobData]) => {
    const assigned = jobData[shift] && jobData[shift][day] ? jobData[shift][day] : [];
    if (assigned.includes(name)) assignedJob = job;
    });
    shifts.push({ shift, job: assignedJob });
    if (assignedJob) dayHours += SHIFT_HRS[shift] || 0;
    });
    const isDayOff = dayHours === 0;
    if (!isDayOff) workingDays++;
    totalHours += dayHours;
    const card = document.createElement('div');
    card.className = 'day-card' + (isDayOff ? ' day-off' : '');
    const header = document.createElement('div');
    header.className = 'day-header';
    header.innerHTML = `<span>${day}</span>${!isDayOff ? `<span class="day-hours">${dayHours}h</span>` : ''}`;
    card.appendChild(header);
    if (isDayOff) {
    const offLabel = document.createElement('div');
    offLabel.className = 'day-off-label';
    offLabel.textContent = 'Day off';
    card.appendChild(offLabel);
    } else {
    shifts.forEach(({ shift, job }) => {
    const row = document.createElement('div');
    row.className = 'shift-row';
    const timeEl = document.createElement('div');
    timeEl.className = 'shift-time';
    timeEl.textContent = shift;
    row.appendChild(timeEl);
    if (job) {
    const jobEl = document.createElement('div');
    jobEl.className = 'shift-job';
    jobEl.style.background = jobColour(job);
    jobEl.textContent = job;
    row.appendChild(jobEl);
    const hrsEl = document.createElement('div');
    hrsEl.className = 'shift-hours';
    hrsEl.textContent = SHIFT_HRS[shift] + 'h';
    row.appendChild(hrsEl);
    } else {
    const offEl = document.createElement('div');
    offEl.className = 'shift-off';
    offEl.textContent = '—';
    row.appendChild(offEl);
    }
    card.appendChild(row);
    });
    }
    container.appendChild(card);
    });
    document.getElementById('total-hours').textContent = totalHours;
    document.getElementById('working-days').textContent = workingDays;
    }
    load();
    </script>
